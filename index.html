import numpy as np
import pandas as pd
from sklearn.ensemble import RandomForestRegressor
from sklearn.model_selection import train_test_split, GridSearchCV
from sklearn.preprocessing import StandardScaler
from sklearn.metrics import r2_score, mean_absolute_error
from datetime import datetime
 
# ── 1. Cargar datos históricos ────────────────
df = pd.read_csv('consumos_historicos_2024.csv')
>>> ✔ 4.820 registros cargados
 
# ── 2. Preprocesamiento ──────────────────────
features = ['potencia_kw', 'horas_uso', 'temp_exterior',
            'superficie_m2', 'ocupacion_pct', 'mes']
X = df[features]
y = df['eficiencia_energetica']
scaler = StandardScaler()
X_scaled = scaler.fit_transform(X)
X_train, X_test, y_train, y_test = train_test_split(
    X_scaled, y, test_size=0.20, random_state=42)
>>> ✔ Train: 3.856 | Test: 964 muestras
 
# ── 3. Optimización hiperparámetros ─────────
param_grid = {
    'n_estimators': [100, 200, 300],
    'max_depth': [8, 12, None],
    'min_samples_split': [2, 5]
}
grid_cv = GridSearchCV(RandomForestRegressor(random_state=42),
                      param_grid, cv=5, scoring='r2', n_jobs=-1)
grid_cv.fit(X_train, y_train)
>>> ✔ Best: n_estimators=200, max_depth=12
 
# ── 4. Entrenar modelo final ─────────────────
model = grid_cv.best_estimator_
y_pred = model.predict(X_test)
r2 = r2_score(y_test, y_pred)
mae = mean_absolute_error(y_test, y_pred)
>>> R² Score : 0.9527
>>> MAE : 1.83%
 
# ── 5. Predicción proyecto HVAC ─────────────
nuevo = {'potencia_kw':150, 'horas_uso':10,
         'temp_exterior':25, 'superficie_m2':3500,
         'ocupacion_pct':85, 'mes':datetime.now().month}
X_new = scaler.transform(pd.DataFrame([nuevo]))
ef = model.predict(X_new)[0]
>>> Eficiencia estimada: 94.7%
 
# ── 6. Ahorro económico anual ────────────────
consumo_base = 150 * 10 * 365
ahorro_kwh = consumo_base * (1 - ef / 100)
ahorro_usd = ahorro_kwh * 0.15
>>> Ahorro: 29.468 kWh → USD 4.420/año
 
# ── 7. Exportar a BIM Revit ──────────────────
from bim_ai import BIMGenerator, RevitExporter
gen = BIMGenerator(model=model, standard='IEC-60364')
bim = gen.create(specs=nuevo)
RevitExporter.save(bim, 'proyecto_hvac_2026.rvt')
>>> ✔ Modelo BIM exportado a Revit
 
# ── 8. Importancia variables ─────────────────
imp = pd.Series(model.feature_importances_,
                index=features).sort_values(ascending=False)
>>> potencia_kw 0.3812
>>> temp_exterior 0.2240
>>> horas_uso 0.1958
>>> ocupacion_pct 0.1102
 
# ── 9. Serializar para producción ────────────
import joblib
joblib.dump(model, 'nahuel_model_v2.4.pkl')
joblib.dump(scaler, 'nahuel_scaler_v2.4.pkl')
